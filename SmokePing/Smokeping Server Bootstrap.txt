#!/bin/bash

sudo apt-get update && sudo apt-get install smokeping
cd /etc/smokeping/config.d
sudo curl https://raw.githubusercontent.com/jacktooandroid/smokeping/master/SmokePing/Probes.txt -o Probes
sudo wget https://raw.githubusercontent.com/jacktooandroid/smokeping/master/SmokePing/Probes.txt -O Probes
sudo curl https://raw.githubusercontent.com/jacktooandroid/smokeping/master/SmokePing/Presentation.txt -o Presentation
sudo wget https://raw.githubusercontent.com/jacktooandroid/smokeping/master/SmokePing/Presentation.txt -O Presentation
sudo curl https://raw.githubusercontent.com/jacktooandroid/smokeping/master/SmokePing/Targets.txt -o Targets
sudo wget https://raw.githubusercontent.com/jacktooandroid/smokeping/master/SmokePing/Targets.txt -O Targets
sudo service smokeping restart

cd /var/www/html/
sudo cp index.html index.html1
sudo curl https://raw.githubusercontent.com/jacktooandroid/smokeping/master/SmokePing/Redirection%20to%20SmokePing.html -o index.html
sudo wget https://raw.githubusercontent.com/jacktooandroid/smokeping/master/SmokePing/Redirection%20to%20SmokePing.html -O index.html
sudo service apache2 restart
